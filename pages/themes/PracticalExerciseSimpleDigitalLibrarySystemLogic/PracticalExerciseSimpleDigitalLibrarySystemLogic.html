Introduction

In this exercise, we'll create the logic for a simple digital library system using JavaScript.
The digital library system will allow users to add books, display the list of books, and search for books by title or author.

Setup Project

Create a new folder for your project (DigitalLibrary) and 2 files: index.html and main.js
Make sure you have the following structure:

                DigitalLibrary/
                ├── index.html
                └── main.js
Set up the basic HTML structure.

                <!-- index.html -->
                <!DOCTYPE html>
                <html lang="en">
                <head>
                    <meta charset="UTF-8">
                    <meta name="viewport" content="width=device-width, initial-scale=1.0">
                    <title>Digital Library</title>
                </head>
                <body>
                    <div id="library-container"></div>
                    <input type="text" id="book-title" placeholder="Book Title">
                    <input type="text" id="book-author" placeholder="Book Author">
                    <button id="add-book">Add Book</button>
                    <input type="text" id="search-term" placeholder="Search by Title or Author">
                    <button id="search-book">Search</button>
                    <div id="search-results"></div>
                    <script src="main.js"></script>
                </body>
                </html>

Define Book Structure

Define the book structure in your JavaScript file.
Each Book object should contain a title and an author.
Use an array to store book objects.
Or use a Library object which will hold all books and the methods to add, search and display them.

                // main.js

                // Book constructor
                function Book(title, author) {
                    this.title = title;
                    this.author = author;
                }

                // Array to store books
                let books = [];

Add Books

Add functionality to add books to the library.
Use event listeners to capture user input and add new book objects to the books array.

                let addBookButton = document.getElementById('add-book');

                addBookButton.addEventListener('click', function() {
                    let title = document.getElementById('book-title').value;
                    let author = document.getElementById('book-author').value;

                    if (title && author) {
                        let newBook = new Book(title, author);
                        books.push(newBook);
                        displayBooks();
                    }
                });

                function displayBooks() {
                    let libraryContainer = document.getElementById('library-container');
                    libraryContainer.innerHTML = '';

                    for (let i = 0; i < books.length; i++) {
                        libraryContainer.innerHTML += `
                            

                                
${books[i].title}

                                
${books[i].author}

                            

                        `;
                    }
                }

Search Books

Add functionality to search for books by title or author.
Tip: you can use the Array includes method to find whether an array includes a certain value among its entries. You can implement your own array filtering using loop, or to use the Array filter method.
Filter the books array based on the search term and display the results.

                let searchButton = document.getElementById('search-book');

                searchButton.addEventListener('click', function() {
                    let searchTerm = document.getElementById('search-term').value.toLowerCase();
                    let searchResults = document.getElementById('search-results');
                    searchResults.innerHTML = '';

                    let filteredBooks = books.filter(book =>
                        book.title.toLowerCase().includes(searchTerm) ||
                        book.author.toLowerCase().includes(searchTerm)
                    );

                    for (let i = 0; i < filteredBooks.length; i++) {
                        searchResults.innerHTML += `
                            

                                
${filteredBooks[i].title}

                                
${filteredBooks[i].author}

                            

                        `;
                    }
                });
